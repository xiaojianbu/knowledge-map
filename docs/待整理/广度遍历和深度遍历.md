深度优先遍历以深度为主，即遍历完某一个节点之后，才会继续往下遍历兄弟节点，这个只需要循环遍历就行了。
function dfs (node) {
printInfo(node)
if(node.children.length){
Array.from(node.children).forEach(function (el) {
dfs(el)
})
}
}
